
修改中断

```c
if (__HAL_UART_GET_FLAG(&huart1, UART_FLAG_IDLE) != RESET)//触发idle中断
  {
      __HAL_UART_CLEAR_IDLEFLAG(&huart1);//清除标志位

      HAL_UART_DMAStop(&huart1);//停止转运
      uint16_t len = RX_BUF_SIZE - __HAL_DMA_GET_COUNTER(huart1.hdmarx);//获取转运数量

    //----------读取(帧解析/buf缓存)------------
      if (len > 0 && len <= RX_BUF_SIZE)
      {
          for (uint16_t i = 0; i < len; i++)
          {
              uart_rx_parse_buf[i] = rx_buf[i];
          }
          uart_rx_len = len;
          uart_rx_ready = 1;
      }
    //-----------------------------------------

      HAL_UART_Receive_DMA(&huart1, rx_buf, RX_BUF_SIZE);//重新开始
  }

  /* USER CODE END USART1_IRQn 0 */
  HAL_UART_IRQHandler(&huart1);
```

```c
HAL_StatusTypeDef USART_SendString(char *str, uint32_t timeout)
{
    uint16_t len = 0;
    char *ptr = str;
    
    // 计算字符串长度
    while (*ptr++ != '\0') {
        len++;
    }
    
    return HAL_UART_Transmit(&huart1, (uint8_t*)str, len, timeout);
}
```

## CAN

GM6020

- M3 ID 0x206
- M2 ID 0x208

```C
//过滤器设置
void Can1_Filter_Init(void)
{
  /* Configure CAN filter */
  CAN_FilterTypeDef sFilterConfig;

  sFilterConfig.FilterBank = 0;
  sFilterConfig.FilterMode = CAN_FILTERMODE_IDMASK;
  sFilterConfig.FilterScale = CAN_FILTERSCALE_32BIT;
  sFilterConfig.FilterIdHigh = 0x0000;
  sFilterConfig.FilterIdLow = 0x0000;
  sFilterConfig.FilterMaskIdHigh = 0x0000;
  sFilterConfig.FilterMaskIdLow = 0x0000;
  sFilterConfig.SlaveStartFilterBank = 14;
  sFilterConfig.FilterFIFOAssignment = CAN_FILTER_FIFO0;
  sFilterConfig.FilterActivation = ENABLE;

  HAL_CAN_ConfigFilter(&hcan_motor, &sFilterConfig);
  HAL_CAN_Start(&hcan_motor);
    
  //开启接收中断，注意使用的中断要跟配置的fifo对应
  HAL_CAN_ActivateNotification(&hcan_motor,CAN_IT_RX_FIFO0_MSG_PENDING);
}
```

```c
//发送
void Motor_Control(uint32_t id,int16_t motor1,int16_t motor2,int16_t motor3,int16_t motor4)
{
    CAN_TxHeaderTypeDef TxHeader = {0};
    uint32_t TxMailbox;

    TxHeader.StdId = id;
    TxHeader.IDE = CAN_ID_STD;
    TxHeader.RTR = CAN_RTR_DATA;
    TxHeader.DLC = 0x08;

    can_send[0] = (motor1>>8) & 0xFF;
    can_send[1] = motor1 & 0xFF;
    can_send[2] = (motor2)>>8 & 0xFF;
    can_send[3] = motor2 & 0xFF;
    can_send[4] = (motor3)>>8 & 0xFF;
    can_send[5] = motor3 & 0xFF;
    can_send[6] = (motor4>>8) & 0xFF;
    can_send[7] = motor4 & 0xFF;

    tx_state = HAL_CAN_AddTxMessage(&hcan_motor, &TxHeader, can_send, &TxMailbox);

    tx_free = HAL_CAN_GetTxMailboxesFreeLevel(&hcan_motor);
    can_err = HAL_CAN_GetError(&hcan_motor);
    can_tsr = hcan_motor.Instance->TSR;
    can_esr = hcan_motor.Instance->ESR;
    can_msr = hcan_motor.Instance->MSR;
}
```

```c
//接收
void Motor_Feedback(void)
{
    CAN_RxHeaderTypeDef RxHeader;
    uint8_t rxData[8];

    HAL_CAN_GetRxMessage(&hcan_motor, CAN_RX_FIFO0, &RxHeader, rxData);

    if(RxHeader.StdId == M2_ID)
    {
        M3.angle=(rxData[0] << 8) | rxData[1];
        M3.speed=(rxData[2] << 8) | rxData[3];
        M3.current=(rxData[4] << 8) | rxData[5];
        M3.temperature=rxData[6];
    }
}
```

### 注意事项

- 使用can的时候因为电机需要有相应差分信号，板子也需要24V供电
- can引脚不是默认，以RM官方为准
- 电机的终端电阻
- 电机不转的时候可以先按下板子的rst，刷新寄存器状态，可能是因为错误状态导致卡在错误中断了
- can1 是PD0、PD1
