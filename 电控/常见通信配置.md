## uart

配置完cube之后需要在初始化代码中加入这些

```c
__HAL_UART_CLEAR_IDLEFLAG(&huartx);//清除空闲标志位
__HAL_UART_ENABLE_IT(&huartx, UART_IT_IDLE);//开启空闲中断
HAL_UART_Receive_DMA(&huartx, rx_buf, RX_BUF_SIZE);//开启dma接收
```

修改中断

```c
if (__HAL_UART_GET_FLAG(&huart1, UART_FLAG_IDLE) != RESET)//触发idle中断
  {
      __HAL_UART_CLEAR_IDLEFLAG(&huart1);//清除标志位

      HAL_UART_DMAStop(&huart1);//停止转运
      uint16_t len = RX_BUF_SIZE - __HAL_DMA_GET_COUNTER(huart1.hdmarx);//获取转运数量

    //----------读取(帧解析/buf缓存)------------
      if (len > 0 && len <= RX_BUF_SIZE)
      {
          for (uint16_t i = 0; i < len; i++)
          {
              uart_rx_parse_buf[i] = rx_buf[i];
          }
          uart_rx_len = len;
          uart_rx_ready = 1;
      }
    //-----------------------------------------

      HAL_UART_Receive_DMA(&huart1, rx_buf, RX_BUF_SIZE);//重新开始
  }

  /* USER CODE END USART1_IRQn 0 */
  HAL_UART_IRQHandler(&huart1);
```
