```sh
#!/bin/bash

# 定义清理函数

cleanup() {
    echo "Stopping..."
    kill "$PID_A" "$PID_B" 2>/dev/null
    wait "$PID_A" "$PID_B" 2>/dev/null
    echo "All child processes stopped."
    exit

}

# 捕捉终止信号

trap cleanup SIGINT SIGTERM

# 启动子进程

python3 script_a.py &
PID_A=$!
echo "Started script_a.py with PID $PID_A"

python3 script_b.py &
PID_B=$!
echo "Started script_b.py with PID $PID_B"

# 保持脚本运行，等待子进程结束

wait
```
